[docs]def draw_line(x1, y1, x2, y2):

    if x1 == y1 and x2 == y2:
        GL.glBegin(GL.GL_POINTS)
        GL.glVertex2f(x1, y1)
    else:
        GL.glBegin(GL.GL_LINES)
        GL.glVertex2i(x1, y1)
        GL.glVertex2i(x2, y2)

    GL.glEnd()

[docs]def draw_line_f(x1, y1, x2, y2):

    epsilon = .1

    if (epsilon_float(x1, x2, epsilon) and
        epsilon_float(y1, y2, epsilon)):

        GL.glBegin(GL.GL_POINTS)
        GL.glVertex2f(x1, y1)
        
    else:
        GL.glBegin(GL.GL_LINES)
        GL.glVertex2f(x1, y1)
        GL.glVertex2f(x2, y2)

    GL.glEnd()
